// arrays de bool: criba para detectar pares (true) e impares (false)

func main() {
    var isEven [8]bool;

    var i int = 0;
    for i < 8 {
        isEven[i] = i % 2 == 0;
        i = i + 1;
    }

    var k int = 0;
    for k < 8 {
        if isEven[k] {
            println(k, " es par");
        } else {
            println(k, " es impar");
        }
        k = k + 1;
    }
}
