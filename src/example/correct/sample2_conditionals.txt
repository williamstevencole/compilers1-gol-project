var threshold int = 10;

// clasifica un numero segun varios rangos
func classify(n int) int {
    if n < 0 {
        return -1;
    } else if n == 0 {
        return 0;
    } else if n < threshold {
        return 1;
    } else {
        return 2;
    }
}

/* devuelve el mayor de dos valores
   usando if/else simple */
func max(a int, b int) int {
    if a >= b {
        return a;
    } else {
        return b;
    }
}

func main() {
    var v int = 25;
    category := classify(v); // declaracion corta

    var m int = max(v, threshold);

    print("category: ", category);
    println("max: ", m);
}
